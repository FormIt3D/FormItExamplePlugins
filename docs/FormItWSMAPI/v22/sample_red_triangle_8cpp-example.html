sample_red_triangle.cpp

A simple WSM API usage example that builds a red triangle.

#include "samples.h"
#include <WSM/WSMApi.h>
#include <WSM/GeometryDataContainers.h>
#include <iostream>
WSM::Result sample_red_triangle()
{
std::cout << "\nsample_red_triangle\n" << std::endl;
// Create a history
size_t nHistoryID = WSM::INVALID_ID;
WSM::Result result = WSM::APICreateHistory(nHistoryID);
if (!result.IsGood()) return result;
// Define a Triangle
std::vector<WSM::FaceData> aFaces;
WSM::FaceData face;
face.m_outerLoop.m_aVertices.emplace_back(0, 0, 0);
face.m_outerLoop.m_aVertices.emplace_back(100, 0, 0);
face.m_outerLoop.m_aVertices.emplace_back(100, 100, 50);
face.m_nBackMaterialIndex = 0;
face.m_nFrontMaterialIndex = 0;
aFaces.push_back(face);
// Define a Red Material
std::vector<WSM::MaterialData> aMaterials;
WSM::MaterialData material;
material.m_color = WSM::Color(255, 0, 0, 255);
aMaterials.push_back(material);
// Create geometries
result = WSM::APICreateGeometryFromData(nHistoryID, {}, aMaterials, {}, {}, aFaces, {}, {}, {});
if (!result.IsGood()) return result;
// Get created object IDs
std::vector<size_t> objectIDs;
result = WSM::APIGetAllNonOwnedReadOnly(nHistoryID, objectIDs);
// Write to file
result = WSM::APISaveToFileReadOnly(nHistoryID, objectIDs, "out.sample_red_triangle.wsm");
if (!result.IsGood()) return result;
return result;
}
